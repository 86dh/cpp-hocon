163  std::string msg;
164  for (auto &p : problems_) {
165  auto sep = std::string(": ");
166  for (auto &s : {p.origin->description(), sep, p.path, sep, p.problem, std::string(", ")}) {
167  msg.append(s);
168  }
169  }
170  if (msg.empty()) {

173  msg.resize(msg.length() - 2);
